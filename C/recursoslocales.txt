void inicializar_recursos(){
    recursos_disponibles_nodo.cpu = rand()%8 + 1;
    recursos_disponibles_nodo.mem = 1024 * (rand()%16 + 1);
    recursos_disponibles_nodo.gpu = rand()%5;
}

// CHEQUEA SI HAY RECURSOS SUFICIENTES DE UN DETERMINADO TIPO
int hay_recursos(char* resource,int amount){
    int hay_recurso = 0; 
    if (!strcmp(resource,"cpu')){
        if (recursos_disponibles_nodo.cpu >= amount)
            hay_recursos++; 
    }
    else if (!strcmp(resource,"mem')){
        if (recursos_disponibles_nodo.mem >= amount)
            hay_recursos++;
    }
    else{
        if (recursos_disponibles_nodo.gpu >= amount)
            hay_recursos++;
    }
    return hay_recursos;
}
